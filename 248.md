void fx() {
    int n;
    cin >> n;
    vi tab(n);

    int dp[250][250];
    int res = INT_MIN;

    for(auto &i: tab) cin >> i;

    for(int len = 1; len <= n; len++){
        for(int i = 0; i + len <= n; i++){
            if(len == 1){
                dp[i][i] = tab[i];
                continue;
            }
            
            int j = i + len -1;
            dp[i][j] = -1;
            
            for(int k = i; k < j; k++){
                if(dp[i][k] != -1 && dp[k+1][j] == dp[i][k]){
                    dp[i][j] = max(dp[i][j], dp[i][k] +1);
                }
            }

            res = max(res, dp[i][j]);

        }
    }

    cout << res << endl;
}
