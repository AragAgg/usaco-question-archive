\#include <bits/stdc++.h>

void fx() {
    // Functionality for fx
    ll n, w;
    cin >> n >> w;

    vector<ll>dp(w+1, -1);
    dp[0] = 0;


    for(int i = 0; i < n; i++){
        int weight, value;
        cin >> weight >> value;

        for(int j = w; j >= 0; j--) if(j + weight <= w) dp[j + weight] = max(dp[j + weight], dp[j] + value);
    }

    cout << *max_element(all(dp)) << endl;
}
