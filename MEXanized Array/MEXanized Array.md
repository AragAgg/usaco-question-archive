\#define debug_vector(v) _debug_vector(\#v, v)
template<typename T>
void _debug_vector(const string& name, const vector<T>& a) {
    cerr << name << " = [ ";
    for(const auto &x : a) cerr << x << ' ';
    cerr << "]\n";
}

// I/O redirection for local testing
\#define iofile(io) \
        freopen((io + ".in").c_str(), "r", stdin); \
        freopen((io + ".out").c_str(), "w", stdout);

// Common outputs
void yes() { cout << "YES" << '\n'; }
void no() { cout << "NO" << '\n'; }

void fx() {
    // Functionality for fx
    int n, k, x;
    cin >> n >> k >> x;
    if(k - x >= 2 or k > n){
        cout << -1 << endl;
        return;
    }
    
    int ans = 0;
    ans += ((k-1)*(k))/2;
    
    if(k == x){
        ans += (n-k)*(x-1);
    }else ans += (n-k)*x;
    
    cout << ans << endl;
}

int main() {

    int t; cin >> t;
    while(t--) fx();

}
